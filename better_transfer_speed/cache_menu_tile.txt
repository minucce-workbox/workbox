.org 0x3d8f
.area 0x3dbd - 0x3d8f


/*	Cache tile to RAM	*/


	ld a, (0xd140)			; dst
	ld l, a
	ld h, 0x00

	sla l :: rl h
	sla l :: rl h
	sla l :: rl h
	sla l :: rl h


	ld a, (0xd138)			; rom ptr
	add a, l
	ld l, a

	ld a, (0xd139)
	adc a, h
	ld h, a

	ld de, 0xd031			; dst  (fixed)
	ld b, 0x10


@@z_loop:
	ldi a, (hl)				; src
	ld (de), a				; dst
	inc e

	dec b
	jr nz, @@z_loop

	ret


.endarea
