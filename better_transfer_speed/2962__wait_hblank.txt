.org 0x2962
.area 0x2973 - 0x2962


/*	Wait for next h-blank	*/


	push af



@@z__blank:
	ldh a, (0x41)				; blank time
	and a, 0x03
	jr z, @@z__blank



@@z__drawing:
	ldh a, (0x41)				; active draw
	and a, 0x03
	jr nz, @@z__drawing



	pop af
	ret


.endarea
