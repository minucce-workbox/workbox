;
;  Credit palette fading effect
;


org 0x4adf, 0x06

	ld a, 0x04					; no hud bar
	ldh (0x8f), a

	ld a, 0x80					; white fade-in
	ldh (0x90), a


	ld a, 0x05					; fade 1
	call 0x1dc3

	ld a, 0x05					; fade 2
	call 0x1dc3

	ld a, 0x05					; fade 3
	call 0x1dc3


	jr 0x4b09


freepc 0x4b09
